import pandas as pd
import numpy as np

def triangle_judgement(triangle: pd.DataFrame) -> pd.DataFrame:
    triangle = triangle.assign(triangle=lambda t: np.where(
            (t.x + t.y > t.z) 
            & (t.y + t.z > t.x)
            & (t.z + t.x > t.y)
        , "Yes"
        , "No"
    ))
    return triangle
